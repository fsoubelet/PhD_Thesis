\newcommand{\newglossaryacronym}[4]{
    \newacronym{#1acr}{#2}{\acrlong{#1}}
    \newglossaryentry{#1}{
        name={#2 (#3)},
        text={#2},
        % symbol={#2},
        short={#2},
        long={#3},
        first={#3 (#2)},
        firstplural={#3\glspluralsuffix~(#2\glspluralsuffix)},
        description={#4\glsadd{#1acr}}
    }
}

% ----- Nomenclature ----- %

\newglossaryentry{AC_dipole}{
    name={AC Dipole},
    text={AC dipole},
    description={Dipole magnet attached to an AC power source with variable frequency and strength, allowing to impose forced oscillations on the \gls{beam}. See \cref{subsection:beam_instrumentation_for_optics_measurements}}
}
\newglossaryentry{ampdet}{
    name={amplitude detuning},
    description={Tune change with the transversal amplitude of a particle. See \cref{subsection:amplitude_detuning}}
}
\newglossaryentry{beam}{
    name={beam},
    description={Short for "Particle Beam".  The name for the collection of all particles in an accelerator traveling in the same direction. In the \acrfull{LHC} there are two beams (Beam~\num{1} and Beam~\num{2}) traversing in opposite directions and colliding at the \glspl{IP}}
}
\newglossaryentry{beta-function}{
    name={beta-function},
    text={\(\beta\)-function},
    description={Value of the \gls{Twiss_parameters} \(\beta\) as a function of longitudinal location. This value is closely related to the amplitude $\mathcal{A}$ of the betatron-oscillations and hence the size of the \gls{beam} at that location via the \gls{action} \glssymbol{action} by \(\mathcal{A} = \sqrt{2J\beta}\)}
}
\newglossaryentry{beta-beating}{
    name={beta-beating},
    text={\(\beta\)-beating},
    description={Relative difference of the \glspl{beta-function} between measurement and model: \((\beta_{\mathrm{measured}} - \beta_{\mathrm{model}}) / \beta_{\mathrm{model}}\). It is a good indicator of the quality of the linear optics. See \cref{section:optics_measurements_and_corrections}}
}
\newglossaryentry{betatron_coupling}{
    name={betatron coupling},
    description={Coupling between the horizontal and vertical transverse motion of a particle. Common sources of coupling are \gls{skew} quadrupoles and solenoids. See \cref{section:betatron_coupling}}
}
\newglossaryentry{bunch}{
    name={bunch},
    description={Short for "Particle Bunch". Due to the \acrshort{RF} acceleration scheme, a continuous \gls{beam} is not possible in a synchrotron and particles become bunched}
}
\newglossaryentry{closed_orbit}{
    name={Closed Orbit},
    text={closed orbit},
    description={The orbit of a particle with reference momentum, sometimes also called "reference orbit" or "ideal orbit". See \cref{subsection:transport_and_guiding_of_charged_particles}}
}
\newglossaryentry{courantsnyderparameter}{
    name={Courant-Snyder Parameters},
    user1={Courant-Snyder},
    description={Set of quantities describing the distribution of positions and momenta of the particles in a \gls{beam}. Also known as \gls{Twiss_parameters}. See for instance the \gls{beta-function}}
}
\newglossaryentry{experiment}{
    name={experiment},
    description={In the context of the \gls{LHC}, an "experiment" denotes one of the particle physics detector experiments and its accompanying data-collecting research structure. The four largest detectors, \acrshort{ATLAS}, \acrshort{CMS}, \acrshort{ALICE}, \acrshort{LHCb} are located at the four colliding \glspl{IP} of the \gls{LHC}}
}
\newglossaryentry{feed-down}{
    name={feed-down},
    description={Particles passing off-center through a multipole field experience effects akin to the influence of lower-order multipoles due to the orbit offset}
}
\newglossaryentry{flat_optics}{
    name={flat optics},
    description={\Gls{optics} in which the \glstext{betastar} is different for both transverse planes, usually much smaller in one than in the other. See also \glstext{round_optics}}
}
\newglossaryentry{hadron}{
    name={hadron},
    description={Composite subatomic particles consisting of two or more quarks, held together by strong interactions}
}
\newglossaryentry{knob}{
    name={knob},
    description={A group setting of magnets powerings to be changed together}
}
\newglossaryentry{longshutdown}{
    name={Long Shutdown},
    description={Planned shutdown periods of the \gls{LHC} between \Glspl{run} spanning multiple years, as opposed to the \gls{YETS}, which only lasts a few months. This time is used for repairs and upgrades of the machine, as well as general maintenance}
}
\newglossaryentry{luminosity}{
    name={luminosity},
    description={The ratio of the number of events, i.e. collisions, detected per cross section, either per time interval ("instantaneous luminosity") or in total ("integrated luminosity", e.g. since the beginning of a year, a \gls{run} or of operation). See \cref{section:luminosity}}
}
\newglossaryentry{MADX}{

}
\newglossaryentry{normal}{
    name={normal magnets},
    text={normal},
    description={Magnet with a "normal" oriented multipole field with pole-tips off the horizontal axis. Opposed to a \gls{skew} magnet of the same order \(n\), for which the field is rotated by \(\pi / (2n)\)~\unit{\radian}}
}
\newglossaryentry{omc3}{
    name={omc3},
    text={omc3},
    description={Python package maintained and applied by the \gls{OMC} team to analyse measurements from different accelerators and calculate corrections. See~\cite{CODE:OMC:omc3}}
}
\newglossaryentry{optics}{
    name={optics},
    description={Refers to accelerator \gls{beam} optics, describing the particle motion through an accelerator as defined by the elements (i.e. magnets) of the machine. As the behaviour of a particle beam in magnetic fields shows similarities to a light beam propagating through lenses and can be described with similar equations, a lot of theory and nomenclature has been borrowed from ray optics}
}
\newglossaryentry{round_optics}{
    name={round optics},
    description={\Gls{optics} in which the \glstext{betastar} is identical for both planes. See also: \gls{flat_optics}} 
}
\newglossaryentry{run}{
    name={run},
    description={Consecutive years of \gls{LHC} operation, separated by long shutdowns. \Gls{run}~\num{1} has taken place from \num{2008} to \num{2013}, \Gls{run}~\num{2} from \num{2015} to \num{2018} and \Gls{run}~\num{3} started in \num{2022}}
}
\newglossaryentry{skew}{
    name={skew magnets},
    text={skew},
    description={Magnet with a "skew" oriented multipole field, with pole-tips on the horizontal axis. Opposed to a \gls{normal} magnet of the same order \(n\), for which the field is rotated by \(\pi / (2n)\)~\unit{\radian}}
}
\newglossaryentry{synchronous}{
    name={synchronous particle},
    text={synchronous},
    description={The ideal particle that defines the reference trajectory in a circular accelerator. See \cref{subsection:transport_and_guiding_of_charged_particles}}
}
\newglossaryentry{trim}{
    name={trim},
    description={A trim refers to powering adjustements of a given electrical circuit powering one or several magnets in series or according to a given scheme}
}
\newglossaryentry{triplet}{
    name={triplet},
    description={Name for the final focusing quadrupoles in the \gls{IR} left and right of an \gls{IP}. In principle these consist of three magnets Q\num{1} to Q\num{3} (starting with Q1 closest to the \gls{IP}), where Q\num{1} and Q\num{3} have the same orientation (e.g. focusing) and are opposite to Q\num{2} (e.g. defocusing), depending on \gls{beam}, plane and side. In the \gls{LHC} Q\num{2} is split into two magnets, and in the \gls{HL-LHC} Q\num{1} and Q\num{2} will also each be split in two}
}
\newglossaryentry{tune}{
    name={tune},
    description={Number of betatron oscillations per turn in a circular accelerator. See also: \gls{beta-function}}
}
\newglossaryentry{Twiss_parameters}{
    name={Twiss Parameters},
    text={Twiss parameters},
    description={See \gls{courantsnyderparameter}}
}


% ----- Detailed Acronyms, they show up in the nomenclature ----- %

\newglossaryacronym{ALICE}{ALICE}{A Large Ion Collider Experiment}{
    An \gls{experiment} optimized to study heavy-ion collisions at the \gls{LHC}
}
\newglossaryacronym{ATLAS}{ATLAS}{A Toroidal LHC Apparatus}{
    One of two general purpose detectors at the \gls{LHC}. It investigates a wide range of physics, from the search for the Higgs boson to extra dimensions and particles that could make up dark matter. Although it has the same scientific goals as the \acrshort{CMS} \gls{experiment}, it uses different technical solutions and a different magnet-system design
}
\newglossaryacronym{CMS}{CMS}{Compact Muon Solenoid}{
    One of two general purpose detectors at the \gls{LHC}. It investigates a wide range of physics, from the search for the Higgs boson to extra dimensions and particles that could make up dark matter. Although it has the same scientific goals as the \acrshort{ATLAS} \gls{experiment}, it uses different technical solutions and a different magnet-system design
}
\newglossaryacronym{IP}{IP}{Interaction Point}{
    The specific point at which the two counter-rotating \glspl{beam} interact, i.e. their particles collide. In the \gls{LHC} these are located in the center of \glspl{IR} \numlist{1;2;5;8} inside the detectors, or \glspl{experiment}
}
\newglossaryacronym{IR}{IR}{Insertion Region}{
    Straight section between the arcs of a synchrotron, housing larger facilities such as detectors (\glspl{experiment}), acceleration (\acrshort{RF}), etc. In this document the shorthand IR mostly refers to an "Interaction Region", a.k.a. an Insertion Region hosting an \gls{IP} where \glspl{beam} are made to collide
}
\newglossaryacronym{LHCb}{LHCb}{LHC-beauty}{
    A specialized b-physics \gls{experiment}, designed primarily to measure the parameters of CP violation in the interactions of b-\gls{hadron}, heavy particles containing a bottom quark
}
\newglossaryacronym{MD}{MD}{Machine Development}{
    Studies dedicated to understanding and improving the machine, e.g. by trying to reveal error-sources or attempting their correction, or testing possible future procedures and machine configurations. They usually involve \gls{beam}-time, i.e. measurements in the \gls{CCC}
}
\newglossaryacronym{RF}{RF}{Radio Frequency}{
    Shorthand for the acceleration part of a synchrotron, as the frequency of the accelerating electric field is usually in the radio-frequency range (at the \gls{LHC} \(\approx\) \qty{400}{\mega\hertz})
}
\newglossaryacronym{RWS}{RWS}{Rigid Waist Shift}{
    A specific perturbation of the optics in an \gls{IR} that moves all four betatron waists of the beam simultaneously. See \cref{section:rigid_waist_shift_for_local_coupling_correction}
}
