%----------------------------------------------------------------------------------------
%	MATH COMMANDS FOR RECURRENT EXPRESSIONS IN PARAGRAPHS
%----------------------------------------------------------------------------------------

% Command for inserting the beta-star
\newcommand{\betastar}{\(\beta^{\ast}\)}

% Command for inserting the C-
\newcommand{\Cminus}{\(C^{-}\) }
\newcommand{\AbsCminus}{\(\left|C^{-}\right|\) }

% Command for inserting the f1001 and f1010
\newcommand{\foneohone}{\(f_{1001}\) }  % needs the trailing space
\newcommand{\absfoneohone}{\(\left|f_{1001}\right|\) }  % needs the trailing space
\newcommand{\foneohoneoh}{\(f_{1010}\) }  % needs the trailing space
\newcommand{\absfoneohoneoh}{\(\left|f_{1010}\right|\) }  % needs the trailing space

% Command for inserting beta-function
\newcommand{\betafunction}{\(\beta\)-function }  % needs the trailing space
\newcommand{\betafunctions}{\(\beta\)-functions }  % needs the trailing space

% Command for inserting beta-beating
\newcommand{\betabeating}{\(\beta\)-beating }  % needs the trailing space

% Commands for inserting either IP or IR
\newcommand{\IP}{\(\mathrm{IP}\) }  % needs the trailing space
\newcommand{\IPs}{\(\mathrm{IPs}\) }  % needs the trailing space
\newcommand{\IR}{\(\mathrm{IR}\) }  % needs the trailing space
\newcommand{\IRs}{\(\mathrm{IRs}\) }  % needs the trailing space

% Command for inserting DPSI
\newcommand{\DPSI}{\(\mathrm{DPSI}\) }  % needs the trailing space

%----------------------------------------------------------------------------------------
%	MATH COMMANDS FOR RECURRENT EXPRESSIONS IN MATH MODE
%----------------------------------------------------------------------------------------

% Command for the absolute value of something
\newcommand{\abs}[1]{\left\lvert #1 \right\rvert}
